            });

            print('1 pass $v');

            });

            setState(() {
              favoriteOnHover = false;
              badOnHover = false;
            });
          }

          if (flag == 0) Icon(Icons.favorite, color: favoriteOnHover ? Colors.red : MainContentModel.iconColor),
